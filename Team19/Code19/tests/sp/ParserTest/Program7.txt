procedure nestedIfWhile {
	while (x == y) {
		if (x == y) then {
			while(x > y) {
				read x;
			}

			while (y > x) {
				read y;
			}
		} else {
			if (x != y) then {						// 7
				while(!(x == y)) {
					read y;
				}
			} else {
				while (x >= y) {
					read y;
				}
			}

			if (x >= y) then {
				while(x <= y){
					read y;
				}
			} else {
				read x;
			}
		}
	}
}
