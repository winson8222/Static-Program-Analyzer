procedure nestedIfWhile {
	while (x == y) {
		if (x == y) then {
			while(x > y) {
				read x;
			}

			while (y > x) {
				read y;
			}
		} else {
			if (x != y) then {
				if (x >= y) then {
					while(x <= y){
						read y;
					}
				} else {
					read x;
				}
			} else {
				while (x >= y) {
					read y;
				}
			}
		}
	}
}
